        document.getElementById('enter-button').addEventListener('click', function() {
            const input = document.getElementById('handle-input');
            const logo = document.getElementById('main-logo-container');
            const form = document.getElementById('login-form');
            
            if(input.value.trim() === '') {
                // Shake effect for empty input
                form.classList.add('animate-pulse');
                setTimeout(() => form.classList.remove('animate-pulse'), 500);
                return;
            }

            // Simulate the "Logo Explosion" transition mentioned in the notes
            logo.classList.add('logo-explosion-active');
            form.style.opacity = '0';
            form.style.transform = 'translateY(50px)';
            
            // In a real app, we would redirect here
            console.log('Transitioning to dashboard for:', input.value);
            
            setTimeout(() => {
                alert('Welcome to Family Clash 2026, ' + input.value + '! Preparing your stadium...');
                location.reload(); // Reset for demo purposes
            }, 800);
        });

        // Add focus effect sound logic placeholder
        const inputField = document.getElementById('handle-input');
        inputField.addEventListener('focus', () => {
            // Subtle visual buzz
            inputField.parentElement.classList.add('scale-[1.01]');
        });
        inputField.addEventListener('blur', () => {
            inputField.parentElement.classList.remove('scale-[1.01]');
        });

        // Background parallax movement
        document.addEventListener('mousemove', (e) => {
            const x = (window.innerWidth / 2 - e.pageX) / 50;
            const y = (window.innerHeight / 2 - e.pageY) / 50;
            document.getElementById('main-logo-container').style.transform = `translate(${x}px, ${y}px)`;
        });

<script>
        // Atmospheric Particle Effect
        const canvas = document.getElementById('particle-canvas');
        const ctx = canvas.getContext('2d');
        let particles = [];

        function resize() {
            canvas.width = window.innerWidth;
            canvas.height = window.innerHeight;
        }

        class Particle {
            constructor() {
                this.reset();
            }
            reset() {
                this.x = Math.random() * canvas.width;
                this.y = canvas.height + 10;
                this.size = Math.random() * 2 + 1;
                this.speed = Math.random() * 1 + 0.5;
                this.opacity = Math.random() * 0.5 + 0.1;
                this.color = Math.random() > 0.5 ? '#ff00ff' : '#2ff801';
            }
            update() {
                this.y -= this.speed;
                if (this.y < -10) this.reset();
            }
            draw() {
                ctx.globalAlpha = this.opacity;
                ctx.fillStyle = this.color;
                ctx.beginPath();
                ctx.arc(this.x, this.y, this.size, 0, Math.PI * 2);
                ctx.fill();
            }
        }

        function init() {
            resize();
            for (let i = 0; i < 50; i++) {
                particles.push(new Particle());
            }
        }

        function animate() {
            ctx.clearRect(0, 0, canvas.width, canvas.height);
            particles.forEach(p => {
                p.update();
                p.draw();
            });
            requestAnimationFrame(animate);
        }

        window.addEventListener('resize', resize);
        init();
        animate();
    </script>

<script>
        let currentMultiplier = 1.55;
        let activeTeam = 'A';

        function selectTeam(team, multiplier) {
            activeTeam = team;
            currentMultiplier = multiplier;
            
            // UI Updates
            const btnA = document.getElementById('btn-team-a');
            const btnB = document.getElementById('btn-team-b');
            
            if(team === 'A') {
                btnA.classList.add('active', 'border-primary', 'ring-2', 'ring-primary/20');
                btnB.classList.remove('active', 'border-secondary', 'ring-2', 'ring-secondary/20');
                btnA.querySelector('span:first-child').classList.add('text-primary');
                btnB.querySelector('span:first-child').classList.remove('text-secondary');
            } else {
                btnB.classList.add('active', 'border-secondary', 'ring-2', 'ring-secondary/20');
                btnA.classList.remove('active', 'border-primary', 'ring-2', 'ring-primary/20');
                btnB.querySelector('span:first-child').classList.add('text-secondary');
                btnA.querySelector('span:first-child').classList.remove('text-primary');
            }

            calculatePayout();
        }

        function calculatePayout() {
            const betInput = document.getElementById('bet-amount');
            const payoutDisplay = document.getElementById('payout-amount');
            const multiplierTag = document.getElementById('multiplier-tag');
            
            let val = parseFloat(betInput.value) || 0;
            let payout = (val * currentMultiplier).toFixed(2);
            
            payoutDisplay.innerText = payout;
            multiplierTag.innerText = `${val} x ${currentMultiplier}x`;
            
            // Simple micro-interaction
            payoutDisplay.classList.add('scale-110');
            setTimeout(() => payoutDisplay.classList.remove('scale-110'), 100);
        }

        // Initialize state
        selectTeam('A', 1.55);
    </script>
